## 14.1 tree-sitter 解析器骨架（parser.js）

**性能要点**：`Parser` 实例和语言对象在模块加载时创建一次，`parseFile` 多次调用时直接复用。
文件字节只读一次——先算 SHA256，再 `toString('utf8')` 解析，避免二次 I/O（TOCTOU-safe）。

```javascript
const Parser   = require('tree-sitter');
const path     = require('node:path');
const { createHash } = require('node:crypto');
const fs       = require('node:fs');

// ── B6: 敏感文件正则（解析前拦截，路径永不入图/FTS）──────────────────────
const SENSITIVE_PATTERNS = [
  /\.env(\.|$)/i,
  /\.(pem|key|crt|p12|pfx)$/i,
  /credentials?|secrets?|token/i,
  /id_rsa|id_ed25519|id_ecdsa/i,
  /\.(netrc|pgpass)$/i,
];

// ── 语言映射：模块级初始化，避免 require 在热路径重复执行 ─────────────────
// 注：此处为 v1 基础版本（12 种语言）。
// §14.7.1 给出完整版（15 种，含 Swift/ObjC/Kotlin）+ EXT 别名/`.h` 路由策略。
// 14.1 的全局集合是过渡 fallback，最终抽取以 §14.7.1 `LANG_SPEC[lang]` 为准。
const LANG_LOADERS = {
  js:  () => require('tree-sitter-javascript'),
  jsx: () => require('tree-sitter-javascript'),
  ts:  () => require('tree-sitter-typescript').typescript,
  tsx: () => require('tree-sitter-typescript').tsx,
  py:  () => require('tree-sitter-python'),
  go:  () => require('tree-sitter-go'),
  java:() => require('tree-sitter-java'),
  rs:  () => require('tree-sitter-rust'),
  rb:  () => require('tree-sitter-ruby'),
  php: () => require('tree-sitter-php').php,
  c:   () => require('tree-sitter-c'),
  cpp: () => require('tree-sitter-cpp'),
};
const LANG_CACHE = new Map();   // ext → Language（懒加载后缓存）
const PARSER_POOL = new Map();  // ext → Parser（每语言一个实例）

function getParser(ext) {
  if (PARSER_POOL.has(ext)) return PARSER_POOL.get(ext);
  if (!LANG_LOADERS[ext]) return null;
  if (!LANG_CACHE.has(ext)) {
    try {
      LANG_CACHE.set(ext, LANG_LOADERS[ext]());
    } catch {
      return null; // 语言包未安装或加载失败时降级为 unsupported_lang
    }
  }
  const p = new Parser();
  p.setLanguage(LANG_CACHE.get(ext));
  PARSER_POOL.set(ext, p);
  return p;
}

// 扩展名别名：避免 `.mm` / `.hpp` 等直接落空
const EXT_ALIASES = {
  mm: 'm',
  hpp: 'cpp',
  cc: 'cpp',
  cxx: 'cpp',
};

function classifyHeaderLanguage(filePath, source) {
  if (/@interface|@protocol|@implementation/.test(source)) return 'm'; // ObjC header
  const base = filePath.replace(/\.h$/i, '');
  if (fs.existsSync(`${base}.mm`) || fs.existsSync(`${base}.cpp`)) return 'cpp';
  return 'c';
}

function resolveLanguageForFile(filePath, source) {
  const raw = path.extname(filePath).slice(1).toLowerCase();
  if (!raw) return null;
  if (raw === 'h') return classifyHeaderLanguage(filePath, source);
  return EXT_ALIASES[raw] ?? raw;
}

// ── symbol_key：节点稳定标识，用于跨 rerun 去重 ──────────────────────────
// 格式：<file_path>#<kind>#<name>#L<line_start>
function symbolKey(filePath, kind, name, lineStart) {
  return `${filePath}#${kind}#${name}#L${lineStart}`;
}

// ── 测试函数识别 ──────────────────────────────────────────────────────────
const TEST_NAME_RE  = /^(test|it|describe|should|expect|beforeEach|afterEach|beforeAll|afterAll)/i;
const TEST_FILE_RE  = /\.(test|spec)\.[jt]sx?$|__(tests?|specs?)__/;

function isTestNode(name, filePath) {
  return TEST_FILE_RE.test(filePath) || TEST_NAME_RE.test(name);
}

// ── 主解析入口 ────────────────────────────────────────────────────────────
function parseFile(filePath) {
  const basename = path.basename(filePath);
  if (SENSITIVE_PATTERNS.some(re => re.test(basename))) {
    return { skipped: true, reason: 'sensitive' };
  }

  const bytes  = fs.readFileSync(filePath);  // 读一次：TOCTOU-safe
  const sha    = createHash('sha256').update(bytes).digest('hex');
  const source = bytes.toString('utf8');
  const lang   = resolveLanguageForFile(filePath, source);
  const parser = getParser(lang);
  if (!parser) return { skipped: true, reason: 'unsupported_lang' };

  const tree   = parser.parse(source);
  const nodes  = [];
  const edges  = [];

  // 模块节点（文件本身）
  nodes.push({
    symbol_key: symbolKey(filePath, 'module', basename, 1),
    kind: 'module', name: basename, file_path: filePath,
    line_start: 1, line_end: source.split('\n').length,
    language: lang,
  });

  extractDeclarations(tree.rootNode, filePath, lang, source, nodes, edges, null);
  extractImports(tree.rootNode, filePath, lang, edges);

  return { sha, nodes, edges, skipped: false };
}

// ── 声明提取：Function / Class / 嵌套方法 ────────────────────────────────
// 说明：以下全局集合仅作为 fallback。正式实现见 §14.7.1 `LANG_SPEC[lang]`。
const FUNC_TYPES = new Set([
  'function_declaration', 'function_definition',   // JS/Python
  'method_definition', 'method_declaration',        // JS/Java
  'arrow_function',                                 // JS
  'function_item',                                  // Rust
  'func_declaration',                               // Go
]);
const CLASS_TYPES = new Set([
  'class_declaration', 'class_definition',          // JS/Python
  'struct_item', 'impl_item',                       // Rust
  'type_declaration',                               // Go
]);

function extractDeclarations(node, filePath, ext, source, nodes, edges, parentName) {
  for (const child of node.namedChildren) {
    if (FUNC_TYPES.has(child.type)) {
      const nameNode = child.namedChildren.find(c =>
        c.type === 'identifier' || c.type === 'property_identifier' || c.type === 'name'
      );
      if (!nameNode) { extractDeclarations(child, filePath, ext, source, nodes, edges, parentName); continue; }

      const name  = nameNode.text;
      const line  = child.startPosition.row + 1;
      const sKey  = symbolKey(filePath, 'function', name, line);
      const isTest = isTestNode(name, filePath);

      nodes.push({
        symbol_key: sKey, kind: 'function', name, file_path: filePath,
        line_start: line, line_end: child.endPosition.row + 1,
        parent_name: parentName, language: ext, is_test: isTest ? 1 : 0,
      });

      // 函数体内提取 call_expression → CALLS 边（延迟：函数名解析后再二次扫描）
      extractCalls(child, sKey, edges);

      // 递归提取嵌套函数/方法
      extractDeclarations(child, filePath, ext, source, nodes, edges, name);

    } else if (CLASS_TYPES.has(child.type)) {
      const nameNode = child.namedChildren.find(c =>
        c.type === 'type_identifier' || c.type === 'identifier' || c.type === 'name'
      );
      if (!nameNode) { extractDeclarations(child, filePath, ext, source, nodes, edges, parentName); continue; }

      const name = nameNode.text;
      const line = child.startPosition.row + 1;
      nodes.push({
        symbol_key: symbolKey(filePath, 'class', name, line),
        kind: 'class', name, file_path: filePath,
        line_start: line, line_end: child.endPosition.row + 1,
        language: ext,
      });

      // 类体内的方法递归
      extractDeclarations(child, filePath, ext, source, nodes, edges, name);

    } else {
      extractDeclarations(child, filePath, ext, source, nodes, edges, parentName);
    }
  }
}

// ── call_expression 提取 → CALLS 边（callerSymbolKey → callee 名称）────────
// 注意：callee symbol_key 需要在全图写入后的第二步通过名称查找解析
function extractCalls(funcNode, callerSymbolKey, edges) {
  function walk(node) {
    if (node.type === 'call_expression') {
      const fnNode = node.namedChildren[0];  // 被调用表达式
      // member call: obj.method(...)
      const calleeName = fnNode?.type === 'member_expression'
        ? fnNode.namedChildren.find(c => c.type === 'property_identifier')?.text
        : fnNode?.text;
      if (calleeName && !/^[A-Z]/.test(calleeName[0] ?? '')) {  // 排除构造函数调用
        edges.push({ kind: 'calls', source_symbol_key: callerSymbolKey, target_name: calleeName });
      }
    }
    for (const child of node.namedChildren) walk(child);
  }
  walk(funcNode);
}

// ── import / require 提取 → IMPORTS_FROM 边 ──────────────────────────────
function extractImports(rootNode, filePath, ext, edges) {
  const moduleSymKey = symbolKey(filePath, 'module', path.basename(filePath), 1);

  function walk(node) {
    // ES module: import { foo } from './bar'
    // Python:    import os  （dotted_name 子节点，非 string）
    // 两者共享 import_statement 类型，通过子节点类型区分，避免双重命中
    if (node.type === 'import_statement') {
      const src = node.namedChildren.find(c => c.type === 'string');
      if (src) {
        // JS/TS ES module
        edges.push({ kind: 'imports_from', source_symbol_key: moduleSymKey,
                     target_path_raw: src.text.replace(/['"]/g, '') });
      } else {
        // Python bare import（import os / import a.b.c）
        const mod = node.namedChildren.find(c => c.type === 'dotted_name');
        if (mod) edges.push({ kind: 'imports_from', source_symbol_key: moduleSymKey,
                              target_path_raw: mod.text });
      }
    }
    // CommonJS: require('./bar')
    else if (node.type === 'call_expression') {
      const fn = node.namedChildren[0];
      if (fn?.text === 'require') {
        const arg = node.namedChildren.find(c => c.type === 'arguments')
                       ?.namedChildren.find(c => c.type === 'string');
        if (arg) edges.push({ kind: 'imports_from', source_symbol_key: moduleSymKey,
                              target_path_raw: arg.text.replace(/['"]/g, '') });
      }
    }
    // Python: from x import y / from . import y（import_from_statement 是 Python 专属节点类型）
    else if (node.type === 'import_from_statement') {
      const mod = node.namedChildren.find(c => c.type === 'dotted_name' || c.type === 'relative_import');
      if (mod) edges.push({ kind: 'imports_from', source_symbol_key: moduleSymKey,
                            target_path_raw: mod.text });
    }
    for (const child of node.namedChildren) walk(child);
  }
  walk(rootNode);
}
```

## 14.1.5 符号解析：raw ref → canonical edge（graph.js）

> 本节补充"数据流缺环"：§14.1 `extractCalls` 产出 `target_name`、`extractImports` 产出 `target_path_raw`，但 `edges` 表只接受整数 `source_id / target_id`。本节定义如何从 raw ref 解析到 canonical edge。

**在主构建流程中的位置**（`runBuild` 伪代码）：

```
1. 项目类型检测（adapter）
2. 输入收敛（collectInputFiles）
3. 语言推导（detectPresentLanguages）
4. 全量解析（parseFiles → allNodes + rawEdges）     ← §14.1
5. 节点批量写入 DB（upsert by symbol_key）
6. resolveEdges(db, rawEdges)                       ← 本节：raw ref → canonical edge
7. postprocess（communities + flows + FTS）
```

步骤 5 必须在步骤 6 之前完成，因为解析需要按名称/路径查 DB 中已写入的节点 id。

### 优先级规则

**CALLS 边解析（target_name → target_id）**：

1. 同文件同作用域内存在唯一匹配的函数名 → 命中
2. 同文件内任意层级存在唯一匹配 → 命中
3. 全库唯一名称 → 命中
4. 多个候选 → 记录为 `unresolved_call`（不写 edges，不报错）
5. 零个候选 → 丢弃（外部依赖调用，无 node 对应）

**IMPORTS_FROM 边解析（target_path_raw → target_id）**：

1. 相对路径规范化（`./foo` → 同目录 `foo.js/foo/index.js`）→ 查 `nodes WHERE kind='module' AND file_path`
2. 绝对/模块路径：按语言扩展名尝试 `<repoRoot>/<path>.ext`
3. 无法归一化 → 记录为 `unresolved_import`（不写 edges，不报错）

```javascript
// graph.js（节选）

/**
 * 将 parseFile 产出的 rawEdges 解析成 canonical source_id / target_id 并写入 edges 表。
 *
 * @param {import('better-sqlite3').Database} db
 * @param {{ kind: string, source_symbol_key: string,
 *           target_name?: string, target_path_raw?: string }[]} rawEdges
 * @param {string} repoRoot - 用于路径规范化（IMPORTS_FROM）
 * @returns {{ resolved: number, unresolved: number }}
 */
function resolveEdges(db, rawEdges, repoRoot) {
  // 1. 建立 symbol_key → id 和 name → id[] 两张内存索引
  //    （全量一次 SELECT，避免 N 次单行查询）
  const keyToId  = new Map();   // symbol_key  → id
  const nameToIds = new Map();  // name        → id[]
  const pathToId  = new Map();  // file_path   → id（仅 kind='module'）

  for (const { id, symbol_key, name, file_path, kind } of
       db.prepare('SELECT id, symbol_key, name, file_path, kind FROM nodes').all()) {
    keyToId.set(symbol_key, id);
    if (!nameToIds.has(name)) nameToIds.set(name, []);
    nameToIds.get(name).push(id);
    if (kind === 'module') pathToId.set(file_path, id);
  }

  const insertEdge = db.prepare(`
    INSERT OR IGNORE INTO edges (kind, source_id, target_id, source_key, target_key)
    VALUES (?, ?, ?, ?, ?)
  `);

  let resolved = 0, unresolved = 0;

  db.transaction(() => {
    for (const raw of rawEdges) {
      const srcId = keyToId.get(raw.source_symbol_key);
      if (srcId === undefined) { unresolved++; continue; }  // 源节点未入库（不应出现）

      let tgtId;

      if (raw.kind === 'calls' && raw.target_name) {
        const candidates = nameToIds.get(raw.target_name) ?? [];
        if (candidates.length === 1) {
          tgtId = candidates[0];
        } else if (candidates.length > 1) {
          // 同名歧义：暂存 unresolved，后续同文件优先策略可进一步消歧
          unresolved++;
          continue;
        } else {
          continue;  // 0 个候选：外部依赖，丢弃
        }
      }

      else if (raw.kind === 'imports_from' && raw.target_path_raw) {
        tgtId = resolveImportPath(raw.target_path_raw, raw.source_symbol_key, repoRoot, pathToId);
        if (tgtId === undefined) { unresolved++; continue; }
      }

      else { continue; }  // 其他 kind（inherits 等）由各语言专用解析器直接写 symbol_key

      insertEdge.run(raw.kind, srcId, tgtId, raw.source_symbol_key, '');
      resolved++;
    }
  })();

  return { resolved, unresolved };
}

/**
 * 将 import 路径规范化并查找对应 module 节点 id。
 * 支持 JS 相对路径和模块名（Python dotted import 暂仅做 name 匹配降级）。
 *
 * @returns {number|undefined}
 */
function resolveImportPath(targetPathRaw, sourceSymbolKey, repoRoot, pathToId) {
  const sourceFile = sourceSymbolKey.split('#')[0];
  const sourceDir  = path.dirname(sourceFile);

  // 相对路径：./foo / ../foo
  if (targetPathRaw.startsWith('.')) {
    const candidates = [
      path.resolve(sourceDir, targetPathRaw),
      path.resolve(sourceDir, targetPathRaw + '.js'),
      path.resolve(sourceDir, targetPathRaw + '.ts'),
      path.resolve(sourceDir, targetPathRaw, 'index.js'),
      path.resolve(sourceDir, targetPathRaw, 'index.ts'),
    ];
    for (const c of candidates) {
      const rel = path.relative(repoRoot, c);   // 统一为相对路径后查索引
      if (pathToId.has(rel)) return pathToId.get(rel);
    }
    return undefined;
  }

  // Python dotted import / 非相对路径：降级为 name 匹配
  // 取最后一段（a.b.c → c）与 file_path basename 匹配
  const lastName = targetPathRaw.split(/[./]/).pop();
  for (const [filePath, id] of pathToId) {
    if (path.basename(filePath, path.extname(filePath)) === lastName) return id;
  }
  return undefined;
}

module.exports = { resolveEdges };
```

**验收要求**：

1. 同名函数存在时，`crg impact / query` 结果不串线（以 `symbol_key` 断言，非 name）。
2. Import alias / 相对路径 / index module 三类场景各有 fixture，期望边计数正确。
3. `crg stats` 输出包含 `unresolved_edges` 计数，`> 0` 时附说明不影响已解析边的查询。

## 14.2 SHA-based 增量检测（incremental.js）

**性能要点**：

- 用一条 `SELECT file_path, sha256 FROM fingerprints` 批量拉取所有指纹到 `Map`，避免 N 次单行查询。
- 所有写操作打包在单个 `db.transaction` 中——SQLite 每次 commit 有 fsync 开销，批量提交可将 1000 文件的写入从 ~5 s 降至 ~50 ms。

```javascript
const { createHash } = require('node:crypto');
const fs = require('node:fs');

// 读字节后再算哈希，TOCTOU-safe：同一次 readFileSync 的字节同时用于 SHA 和解析
function computeFileSHA(filePath) {
  return createHash('sha256').update(fs.readFileSync(filePath)).digest('hex');
}

/**
 * 对比磁盘文件与 fingerprints 表，返回三类变更集
 * 性能：1 次批量 SELECT + 1 次事务（N 个 upsert + M 个 delete）
 *
 * @param   {Database}  db
 * @param   {string[]}  filePaths  当前磁盘上的全部目标文件
 * @returns {{ changed: string[], unchanged: string[], deleted: string[] }}
 */
function detectChangedFiles(db, filePaths) {
  // 1. 批量拉取现有指纹到内存 Map（1 次 SELECT）
  const stored = new Map(
    db.prepare('SELECT file_path, sha256 FROM fingerprints').all()
      .map(r => [r.file_path, r.sha256])
  );

  const fileSet   = new Set(filePaths);
  const deleted   = [...stored.keys()].filter(p => !fileSet.has(p));
  const changed   = [];
  const unchanged = [];

  // 2. 计算 SHA，与缓存比对
  const toUpsert = [];
  for (const fp of filePaths) {
    const sha = computeFileSHA(fp);
    if (stored.get(fp) !== sha) { changed.push(fp);   toUpsert.push({ fp, sha }); }
    else                        { unchanged.push(fp); }
  }

  // 3. 单事务批量写入（upsert 变更 + delete 已删除）
  const upsert = db.prepare(`
    INSERT INTO fingerprints (file_path, sha256, updated_at)
    VALUES (?, ?, unixepoch())
    ON CONFLICT(file_path) DO UPDATE
      SET sha256 = excluded.sha256, updated_at = excluded.updated_at
  `);
  const del = db.prepare('DELETE FROM fingerprints WHERE file_path = ?');

  db.transaction(() => {
    for (const { fp, sha } of toUpsert) upsert.run(fp, sha);
    for (const fp of deleted)           del.run(fp);
  })();

  return { changed, unchanged, deleted };
}
```

## 14.3 BFS 影响半径（graph.js）

**性能要点**：不在 BFS 每步中查 DB（否则 N 层 × M 节点 = N×M 次查询）。
改为先将全图边加载到内存邻接表（2 次 SELECT），BFS 全程在内存中进行，DB 查询固定为 O(1)。
对于大型仓库（10 万边），邻接表约占 20-50 MB，可接受。

`crg impact` 和 `crg review-context` 共用此函数。

```javascript
/**
 * 预加载全图邻接表（模块级缓存，build 完成后调用一次即可）
 * fwd: source_id → target_id[]   （正向：依赖谁）
 * rev: target_id → source_id[]   （反向：谁依赖我）
 */
function loadAdjacency(db) {
  const FWD_KINDS = new Set(['calls', 'imports_from', 'tested_by']);
  const REV_KINDS = new Set(['calls', 'imports_from']);

  const fwd = new Map();
  const rev = new Map();

  for (const { source_id, target_id, kind } of db.prepare('SELECT source_id, target_id, kind FROM edges').all()) {
    if (FWD_KINDS.has(kind)) {
      if (!fwd.has(source_id)) fwd.set(source_id, []);
      fwd.get(source_id).push(target_id);
    }
    if (REV_KINDS.has(kind)) {
      if (!rev.has(target_id)) rev.set(target_id, []);
      rev.get(target_id).push(source_id);
    }
  }
  return { fwd, rev };
}

/**
 * 双向 BFS：fan_out（正向依赖链）+ blast_radius（反向被依赖方）
 *
 * @param {{ fwd: Map, rev: Map }} adj  - loadAdjacency() 的返回值
 * @param {number}  nodeId              - 起点节点 DB id
 * @param {number}  maxDepth            - 最大展开深度（默认 3）
 * @returns {{ fan_out, blast_radius }} - 每项含 { id, depth, name?, file_path? }
 */
function impactRadius(adj, nodeId, maxDepth = 3) {
  function bfs(startId, neighbors) {
    const result  = [];
    const visited = new Set([startId]);
    let frontier  = [startId];

    for (let d = 1; d <= maxDepth && frontier.length > 0; d++) {
      const next = [];
      for (const nid of frontier) {
        for (const nb of (neighbors.get(nid) || [])) {
          if (!visited.has(nb)) {
            visited.add(nb);
            next.push(nb);
            result.push({ id: nb, depth: d });
          }
        }
      }
      frontier = next;
    }
    return result;
  }

  return {
    fan_out:      bfs(nodeId, adj.fwd),   // 正向：nodeId 依赖的传递闭包
    blast_radius: bfs(nodeId, adj.rev),   // 反向：依赖 nodeId 的传递闭包
  };
}

// 使用示例（crg impact 命令层）：
//   const adj = loadAdjacency(db);           ← 每次 crg impact 调用前执行一次
//   const result = impactRadius(adj, nodeId);
```

## 14.4 执行流检测 + PageRank-like 评分（flows.js）

**性能要点**：

- PageRank 迭代中不查 DB——先将所有 CALLS 边加载到内存邻接表（1 次 SELECT）。
- 20 次迭代 × N 节点全部是 `Map.get/set`，无 I/O。对 1 万函数节点约 5 ms。
- 流路径（BFS 展开）使用同一邻接表，不额外查询。

```javascript
const DAMPING    = 0.85;
const ITERATIONS = 20;

/**
 * 步骤 1：PageRank-like 评分
 * 将 CALLS 边方向的重要度从入口向被调用函数传播
 * 返回 Map<nodeId, score>
 */
function computePageRank(db) {
  // 一次性加载全部 CALLS 边到邻接表
  const callEdges = db.prepare("SELECT source_id, target_id FROM edges WHERE kind = 'calls'").all();
  const outAdj    = new Map();   // source → target[]
  const inAdj     = new Map();   // target → source[]（用于出度计算）
  const nodeSet   = new Set();

  for (const { source_id, target_id } of callEdges) {
    if (!outAdj.has(source_id)) outAdj.set(source_id, []);
    outAdj.get(source_id).push(target_id);
    if (!inAdj.has(target_id)) inAdj.set(target_id, []);
    inAdj.get(target_id).push(source_id);
    nodeSet.add(source_id);
    nodeSet.add(target_id);
  }

  // 补入孤立节点（无边的函数）
  for (const { id } of db.prepare("SELECT id FROM nodes WHERE kind = 'function'").all()) {
    nodeSet.add(id);
  }

  // 初始化 uniform 分
  const N      = nodeSet.size;
  const scores = new Map([...nodeSet].map(id => [id, 1 / N]));

  // 迭代传播
  for (let iter = 0; iter < ITERATIONS; iter++) {
    const next = new Map([...nodeSet].map(id => [id, (1 - DAMPING) / N]));
    for (const [src, targets] of outAdj) {
      const share = (scores.get(src) || 0) * DAMPING / targets.length;
      for (const tgt of targets) {
        next.set(tgt, (next.get(tgt) || 0) + share);
      }
    }
    for (const [id, s] of next) scores.set(id, s);
  }
  return { scores, outAdj, inAdj };
}

/**
 * 步骤 2：识别入口点并展开调用路径
 *
 * 默认入口判据（通用项目）：fan-in = 0（无任何节点 CALLS 它）。
 * iOS / 领域专用项目：通过 options.isEntryPoint 注入策略，替换此默认判据（§14.8.2）。
 *
 * @param {Database} db
 * @param {{ isEntryPoint?: (node: {id:number, name:string, file_path:string, kind:string})
 *            => { isEntry: boolean, entryType: string|null, actionId: string|null } }} [options]
 */
function detectFlows(db, options = {}) {
  const { scores, outAdj, inAdj } = computePageRank(db);

  const allFuncs = db.prepare(
    "SELECT id, name, file_path, kind, extra FROM nodes WHERE kind = 'function'"
  ).all();  // extra 列含 meta.crossplatform，供 classifyEntryPoint Pattern 3 读取

  // 默认策略：fan-in = 0
  const defaultIsEntry = (f) => ({
    isEntry: !inAdj.has(f.id) || inAdj.get(f.id).length === 0,
    entryType: 'fanin_zero',
    actionId: null,
  });

  // 优先使用注入策略（iOS 适配层等），无注入时回退到默认 fan-in=0
  const isEntryPoint = options.isEntryPoint ?? defaultIsEntry;

  const entries = allFuncs
    .map(f => ({ ...f, entry: isEntryPoint(f) }))
    .filter(f => f.entry.isEntry);

  return entries.map(({ id, name, file_path, entry }) => {
    // BFS 展开调用路径（限深度 5，避免环路导致无限展开）
    // 命名 callPath 而非 path，避免与顶层 require('node:path') 同名混淆
    const callPath = [];
    const visited  = new Set([id]);
    const queue    = [[id, 0]];

    while (queue.length) {
      const [nid, d] = queue.shift();
      if (d > 0) callPath.push({ id: nid, depth: d });
      if (d >= 5) continue;
      for (const tgt of (outAdj.get(nid) || [])) {
        if (!visited.has(tgt)) { visited.add(tgt); queue.push([tgt, d + 1]); }
      }
    }

    return {
      entry_id:    id,
      name,
      file_path,
      action_id:   entry.actionId ?? null,   // iOS CTMediator 稳定地址（非 iOS 项目为 null）
      entry_type:  entry.entryType,
      criticality: +((scores.get(id) || 0).toFixed(6)),
      node_count:  callPath.length + 1,
      path: callPath,   // [{ id, depth }, ...]  —— 供 crg flow <name> 展开使用
    };
  }).sort((a, b) => b.criticality - a.criticality);
}
```

## 14.5 3-Pass 社区检测（communities.js）

**性能要点**：

- Pass 2 健康评估使用全图边集（一次性加载到内存 `Set`），避免每个社区分别查 DB。
- 节点 ID 为整数，IN 子句拼接安全（非用户输入）；大型社区（>500 节点）时改用临时表以防 SQL 过长。
- Pass 3 仅对 oversized 社区触发，正常项目（社区 <25% 节点）零开销。

```javascript
const CONTAINER_DIRS  = new Set(['src', 'lib', 'app', 'pkg', 'packages', 'modules', 'source', 'core']);
const OVERSIZED_RATIO = 0.25;
const D_THRESHOLD     = 0.3;
const I_THRESHOLD     = 0.5;
const IN_CLAUSE_LIMIT = 500;  // 超出则用临时表

// ── 工具：安全 IN 查询（大集合用临时表）────────────────────────────────────
function withTempIds(db, ids, fn) {
  if (ids.length <= IN_CLAUSE_LIMIT) {
    return fn(`(${ids.join(',')})`);  // 直接拼接（全为整数，安全）
  }
  // 大集合：写入临时表再 JOIN
  db.prepare('CREATE TEMP TABLE IF NOT EXISTS _tmp_ids (id INTEGER PRIMARY KEY)').run();
  db.prepare('DELETE FROM _tmp_ids').run();
  const ins = db.prepare('INSERT INTO _tmp_ids VALUES (?)');
  db.transaction(() => { for (const id of ids) ins.run(id); })();
  const result = fn('(SELECT id FROM _tmp_ids)');
  db.prepare('DROP TABLE IF EXISTS _tmp_ids').run();
  return result;
}

// ── Pass 1：自适应目录框架 ────────────────────────────────────────────────
// 跳过 CONTAINER_DIRS，按第一个有实质文件的目录层级分组
// 同时为 Pass 2/3 预加载 node→filePath 映射（避免后续重复查询）
function buildDirGroups(db) {
  const files  = db.prepare("SELECT id, file_path FROM nodes WHERE kind = 'module'").all();
  const groups = new Map();                    // dirKey → Set<node_id>
  const nodeFiles = new Map();                 // node_id → file_path（Pass 3 复用）

  for (const { id, file_path } of files) {
    nodeFiles.set(id, file_path);
    const parts = file_path.replace(/\\/g, '/').split('/');
    let depth = 0;
    while (depth < parts.length - 1 && CONTAINER_DIRS.has(parts[depth])) depth++;
    const dir = parts.slice(0, depth + 1).join('/') || '.';
    if (!groups.has(dir)) groups.set(dir, new Set());
    groups.get(dir).add(id);
  }
  return { groups, nodeFiles };
}

// ── Pass 2：健康评估（全局单次边扫描，O(E + N)）────────────────────────────
//
// 原实现 evaluateHealth(nodeIds, allEdgePairs) 对每个社区遍历全量边，O(C×E)。
// 对大型项目（~900 社区 × ~100K 边）= 9000万 次迭代，是性能热点。
//
// 修复：先建 nodeId → communityIdx 反向映射（O(N)），再对所有边扫描一次（O(E)），
// 等价于逐社区计算 intra/inter，但总复杂度 O(N + E) 而非 O(C×E)。
//
// 此函数现由 buildCommunities 在汇总所有 subGroups 后调用，而不是在每个社区内独立调用。
function loadSemanticEdges(db) {
  const SEMANTIC_KINDS = new Set(['calls', 'imports_from', 'tested_by', 'inherits', 'implements']);
  // 注：'tested_by' 边当前解析器尚未生成，此处预留以备后续实现
  return db.prepare('SELECT source_id, target_id, kind FROM edges').all()
    .filter(e => SEMANTIC_KINDS.has(e.kind));
}

function computeHealthFromCounts(intra, inter, nodeCount) {
  const n = nodeCount;
  if (n < 2) return { status: 'isolated', density: 0, independence: 1 };
  const maxEdges    = n * (n - 1) / 2;
  const density     = maxEdges > 0 ? intra / maxEdges : 0;
  const independence = (intra + inter) > 0 ? intra / (intra + inter) : 1;
  let status;
  if      (density >= D_THRESHOLD && independence >= I_THRESHOLD) status = 'healthy';
  else if (density <  D_THRESHOLD && independence >= I_THRESHOLD) status = 'isolated';
  else if (density >= D_THRESHOLD && independence <  I_THRESHOLD) status = 'scattered';  // 最高风险
  else                                                             status = 'fragmented';
  return { status, density: +density.toFixed(3), independence: +independence.toFixed(3) };
}

// ── Pass 3：Oversized 社区 BFS 连通分量精化 ───────────────────────────────
function splitOversized(db, nodeIds, totalNodes, nodeFiles) {
  if (nodeIds.size <= totalNodes * OVERSIZED_RATIO) return [{ nodes: nodeIds }];

  // 加载社区内部边（withTempIds 处理大集合）
  const ids  = [...nodeIds];
  const edges = withTempIds(db, ids, (inExpr) =>
    db.prepare(`
      SELECT source_id, target_id FROM edges
      WHERE source_id IN ${inExpr} AND target_id IN ${inExpr}
        AND kind NOT IN ('contains','defined_in')
    `).all()
  );

  // BFS 连通分量
  const adj = new Map(ids.map(id => [id, []]));
  for (const { source_id, target_id } of edges) {
    adj.get(source_id)?.push(target_id);
    adj.get(target_id)?.push(source_id);
  }
  const visited    = new Set();
  const components = [];
  for (const start of nodeIds) {
    if (visited.has(start)) continue;
    const comp = new Set(), queue = [start];
    while (queue.length) {
      const node = queue.shift();
      if (visited.has(node)) continue;
      visited.add(node); comp.add(node);
      for (const nb of (adj.get(node) || [])) queue.push(nb);
    }
    components.push(comp);
  }
  if (components.length >= 2) return components.map(nodes => ({ nodes }));

  // Fallback：按二级子目录再分组
  const subdirMap = new Map();
  for (const id of nodeIds) {
    const fp    = nodeFiles.get(id) || '';
    const parts = fp.replace(/\\/g, '/').split('/');
    // 取跳过容器层后的第二个目录段（比 Pass 1 多一层）
    let depth = 0;
    while (depth < parts.length - 1 && CONTAINER_DIRS.has(parts[depth])) depth++;
    const subdir = parts.slice(0, depth + 2).join('/') || '.';
    if (!subdirMap.has(subdir)) subdirMap.set(subdir, new Set());
    subdirMap.get(subdir).add(id);
  }
  if (subdirMap.size >= 2) return [...subdirMap.values()].map(nodes => ({ nodes }));

  // 无边界可拆：标记 note，保持完整，不静默截断
  return [{ nodes: nodeIds, note: 'oversized, no split boundary found' }];
}

// ── 主入口 ────────────────────────────────────────────────────────────────
function buildCommunities(db) {
  const totalNodes            = db.prepare('SELECT COUNT(*) AS n FROM nodes').get().n;
  const { groups, nodeFiles } = buildDirGroups(db);          // Pass 1

  // ── Pass 3 + 汇总所有 subGroups ─────────────────────────────────────────
  // 先把所有社区的节点集合汇总，再做单次 O(E) 边扫描（Pass 2）
  // 这样避免了原来 evaluateHealth(nodeIds, allEdgePairs) 在每个社区内各扫描一遍全图边的 O(C×E) 开销
  const allSubGroups = [];   // [{ dir, nodes: Set<id>, note? }]
  for (const [dir, nodeIds] of groups) {
    const subGroups = splitOversized(db, nodeIds, totalNodes, nodeFiles);  // Pass 3
    for (const { nodes, note } of subGroups) {
      allSubGroups.push({ dir, nodes, note });
    }
  }

  // ── Pass 2：单次 O(E) 边扫描，累计每个社区的 intra / inter ──────────────
  // 构建 nodeId → communityIdx 反向映射（O(N)）
  const nodeToIdx = new Map();   // node_id → allSubGroups index
  for (let i = 0; i < allSubGroups.length; i++) {
    for (const nid of allSubGroups[i].nodes) nodeToIdx.set(nid, i);
  }

  const intra = new Array(allSubGroups.length).fill(0);
  const inter = new Array(allSubGroups.length).fill(0);
  for (const { source_id, target_id } of loadSemanticEdges(db)) {
    const si = nodeToIdx.get(source_id);
    const ti = nodeToIdx.get(target_id);
    if (si === undefined && ti === undefined) continue;
    if (si === ti && si !== undefined) {
      intra[si]++;
    } else {
      if (si !== undefined) inter[si]++;
      if (ti !== undefined) inter[ti]++;
    }
  }

  const communities = allSubGroups.map(({ dir, nodes, note }, i) => {
    const health = computeHealthFromCounts(intra[i], inter[i], nodes.size);
    if (note) health.note = note;
    return { dir, node_ids: [...nodes], health };
  });

  // 写入 DB（crg postprocess 调用，单事务）
  const insertCommunity     = db.prepare(`
    INSERT INTO communities (name, file_count, health_status, health_density, health_independence)
    VALUES (?, ?, ?, ?, ?)
  `);
  const updateNodeCommunity = db.prepare('UPDATE nodes SET community_id = ? WHERE id = ?');

  db.transaction(() => {
    // 外键安全顺序：
    //   1. 先将 nodes.community_id 全部 NULL（清除对旧社区行的引用）
    //   2. 再删旧社区（此时无悬空引用，PRAGMA foreign_keys=ON 下不报错）
    //   3. 插入新社区并回填 community_id
    // 若不先 NULL，DELETE FROM communities 在外键开启时会因 nodes 仍引用旧 id 而报错
    db.prepare('UPDATE nodes SET community_id = NULL').run();
    db.prepare('DELETE FROM communities').run();
    for (const c of communities) {
      const { lastInsertRowid: cid } = insertCommunity.run(
        c.dir, c.node_ids.length, c.health.status, c.health.density, c.health.independence
      );
      for (const nid of c.node_ids) updateNodeCommunity.run(cid, nid);
    }

    // Pass 4：将 module 节点的 community_id 传播到同文件的 function/class/method 节点
    // 背景：buildDirGroups 只处理 kind='module' 节点；若不传播，function/class/method
    //       节点的 community_id 始终为 NULL，导致 surprisingConnections 查询
    //       （WHERE sn.community_id IS NOT NULL）返回零结果
    db.prepare(`
      UPDATE nodes
      SET community_id = (
        SELECT m.community_id FROM nodes m
        WHERE m.kind = 'module' AND m.file_path = nodes.file_path
        LIMIT 1
      )
      WHERE kind != 'module' AND community_id IS NULL
    `).run();
  })();

  const byStatus = Object.fromEntries(
    ['healthy','isolated','scattered','fragmented']
      .map(s => [s, communities.filter(c => c.health.status === s).length])
  );
  return { communities, stats: { total: communities.length, by_status: byStatus } };
}
```

## 14.6 多因子惊喜评分 + 神节点过滤（analyze.js）

**性能要点**：

- 节点度数通过一次 `GROUP BY` 预计算到 `Map`，避免 `surprisingConnections` 的 SQL 中使用相关子查询（原来每条边触发 2 次子查询，N 边 = 2N 次额外查询）。
- `godNodes` 用 `fan_in`（被依赖数）代替双向 `LEFT JOIN edges`，语义更精确且避免重复计数。

```javascript
const CONF_WEIGHT     = { Unknown: 3, Inferred: 2, Observed: 1 };
const PERIPHERAL_DEG  = 2;   // src 度数 ≤ 此值视为边缘节点
const HUB_DEG         = 10;  // tgt 度数 ≥ 此值视为枢纽节点

// ── 预计算节点度数（双向，语义边）────────────────────────────────────────
// 返回 Map<node_id, { out_deg, in_deg }>
function buildDegreeMap(db) {
  const degMap = new Map();
  const ensure = id => {
    if (!degMap.has(id)) degMap.set(id, { out_deg: 0, in_deg: 0 });
    return degMap.get(id);
  };
  const SEMANTIC = new Set(['calls','imports_from','tested_by','inherits','implements']);
  for (const { source_id, target_id, kind } of
       db.prepare('SELECT source_id, target_id, kind FROM edges').all()) {
    if (!SEMANTIC.has(kind)) continue;
    ensure(source_id).out_deg++;
    ensure(target_id).in_deg++;
  }
  return degMap;
}

/**
 * 惊喜连接：返回跨社区非结构边，按 score 降序，附 reasons
 *
 * 4 因子评分：
 *   F1 置信度权重   Unknown=3 / Inferred=2 / Observed=1
 *   F2 跨语言       +1（src_lang ≠ tgt_lang）
 *   F3 跨社区       +1（固定，已在查询层过滤）
 *   F4 边缘→枢纽    +1（src 度数 ≤ PERIPHERAL_DEG && tgt in_deg ≥ HUB_DEG）
 */
function surprisingConnections(db, topN = 20) {
  // 1. 预计算度数（1 次 SELECT）
  const degMap = buildDegreeMap(db);

  // 2. 取跨社区非结构边（带节点元数据，无子查询）
  const edges = db.prepare(`
    SELECT e.source_id, e.target_id, e.kind,
           sn.language   AS src_lang, tn.language   AS tgt_lang,
           sn.confidence AS src_conf,
           sn.name       AS src_name, tn.name       AS tgt_name,
           sn.file_path  AS src_file, tn.file_path  AS tgt_file
    FROM edges e
    JOIN nodes sn ON e.source_id = sn.id
    JOIN nodes tn ON e.target_id = tn.id
    WHERE sn.community_id IS NOT NULL
      AND sn.community_id != tn.community_id
      AND e.kind NOT IN ('imports','contains','defined_in')
  `).all();

  // 3. 评分
  return edges
    .map(e => {
      const reasons = [];
      let score = 0;

      // F1: 置信度
      const cw = CONF_WEIGHT[e.src_conf] ?? 1;
      score += cw;
      if (cw === 3) reasons.push('no-AST-evidence');
      else if (cw === 2) reasons.push('inferred-edge');

      // F2: 跨语言
      if (e.src_lang && e.tgt_lang && e.src_lang !== e.tgt_lang) {
        score += 1;
        reasons.push(`cross-language:${e.src_lang}→${e.tgt_lang}`);
      }

      // F3: 跨社区（已过滤，固定 +1）
      score += 1;
      reasons.push('cross-community');

      // F4: 边缘 → 枢纽
      const srcDeg = degMap.get(e.source_id) || { out_deg: 0, in_deg: 0 };
      const tgtDeg = degMap.get(e.target_id) || { out_deg: 0, in_deg: 0 };
      const srcTotal = srcDeg.out_deg + srcDeg.in_deg;
      if (srcTotal <= PERIPHERAL_DEG && tgtDeg.in_deg >= HUB_DEG) {
        score += 1;
        reasons.push('peripheral→hub');
      }

      return {
        source: { id: e.source_id, name: e.src_name, file: e.src_file },
        target: { id: e.target_id, name: e.tgt_name, file: e.tgt_file },
        kind: e.kind, score, reasons,
      };
    })
    .sort((a, b) => b.score - a.score)
    .slice(0, topN);
}

// ── 神节点过滤（B3）──────────────────────────────────────────────────────
// 取 fan_in（被依赖数）最高的实体节点，排除三类噪声
//
// Rule 1: kind = 'module'           → 文件级容器节点，不是真实代码实体
// Rule 2: name LIKE '.%'            → 方法存根（.toString / .valueOf 等）
// Rule 3: fan_in = 0                → 孤立函数，无人依赖，不是枢纽
//
// 使用 fan_in（入度）而非总度数：hub 的特征是"被很多人依赖"
function godNodes(db, topN = 10) {
  return db.prepare(`
    SELECT n.id, n.name, n.kind, n.file_path,
           COUNT(e.id) AS fan_in
    FROM nodes n
    JOIN edges e ON e.target_id = n.id
      AND e.kind IN ('calls','imports_from','inherits','implements')
    WHERE n.kind != 'module'        -- Rule 1: 排除文件容器
      AND n.name NOT LIKE '.%'      -- Rule 2: 排除方法存根
    GROUP BY n.id
    HAVING fan_in > 0               -- Rule 3: 排除孤立函数
    ORDER BY fan_in DESC
    LIMIT ?
  `).all(topN);
}
```

## 14.7 Swift / ObjC / Kotlin 解析器扩展

> **适用场景**：目标仓库包含 `.swift`、`.m/.mm`（Objective-C/ObjC++）、`.h`、`.kt` 文件。本节在 §14.1 `LANG_LOADERS` 基础上补充 grammar 加载、`.h` 路由、`LANG_SPEC[lang]` 抽取规则和调用边提取规则。

### 14.7.1 按需语言检测（基于最终输入集合）

**设计原则**：语言 grammar 加载、输入收敛、项目类型策略是三件独立的事。

- **输入收敛**：由 `git ls-files` + 内置规则 + `.spec-first-graphignore` 得到 `final_inputs`
- **语言检测**：从 `final_inputs` 推导 `present_languages`，不再做第二次全盘扫描
- **策略选择**：检测项目特征（`Podfile.lock` 等）决定入口/社区策略（§14.8）

这样可以保证“索引范围”和“语言检测范围”严格一致，避免双轨口径导致的漏检/误检。

```javascript
// parser.js —— 全量语言注册表（替换 §14.1 的 12 种基础版本，最终使用此版本）
const LANG_LOADERS = {
  // ── Web / Node ─────────────────────────────────────────────────────────────
  js:    () => require('tree-sitter-javascript'),
  jsx:   () => require('tree-sitter-javascript'),
  ts:    () => require('tree-sitter-typescript').typescript,
  tsx:   () => require('tree-sitter-typescript').tsx,
  // ── 通用后端 ──────────────────────────────────────────────────────────────
  py:    () => require('tree-sitter-python'),
  go:    () => require('tree-sitter-go'),
  java:  () => require('tree-sitter-java'),
  rs:    () => require('tree-sitter-rust'),
  rb:    () => require('tree-sitter-ruby'),
  php:   () => require('tree-sitter-php').php,
  c:     () => require('tree-sitter-c'),
  cpp:   () => require('tree-sitter-cpp'),
  // ── iOS / Android / KMP ───────────────────────────────────────────────────
  swift: () => require('tree-sitter-swift'),
  m:     () => require('tree-sitter-objc'),   // Objective-C .m / .mm
  kt:    () => require('tree-sitter-kotlin'),
};

// 统一抽取规范：按语言定义节点类型 / 名称字段优先级 / 调用导入提取器
const LANG_SPEC = {
  __default: {
    functionTypes: FUNC_TYPES,
    classTypes: CLASS_TYPES,
    nameFields: ['identifier', 'property_identifier', 'name', 'type_identifier'],
    extractCalls: extractCalls,
    extractImports: extractImports,
  },
  swift: {
    functionTypes: new Set(['function_declaration']),
    classTypes: new Set(['class_declaration', 'struct_declaration', 'protocol_declaration']),
    nameFields: ['simple_identifier', 'type_identifier'],
    extractCalls: extractSwiftCalls,
    extractImports: extractSwiftImports,
  },
  m: {
    functionTypes: new Set(['method_definition', 'class_method_definition', 'function_definition']),
    classTypes: new Set(['implementation_definition']),
    nameFields: ['method_selector', 'identifier', 'type_identifier'],
    extractCalls: extractObjcCalls,    // 含 message_expression
    extractImports: extractObjcImports // 含 #import / #include
  },
  kt: {
    functionTypes: new Set(['function_declaration']),
    classTypes: new Set(['class_declaration', 'object_declaration']),
    nameFields: ['simple_identifier', 'type_identifier'],
    extractCalls: extractKotlinCalls,
    extractImports: extractKotlinImports,
  },
};

/**
 * 从输入收敛后的最终输入集合推导语言集合，不再扫描 repoRoot。
 *
 * @param {string[]} finalInputs - 输入收敛链路输出的最终文件列表
 * @returns {Set<string>} presentLanguages
 */
function detectPresentLanguages(finalInputs) {
  const found = new Set();
  for (const filePath of finalInputs) {
    // resolveLanguageForFile 负责 .mm 别名和 .h 启发式路由
    const source = path.extname(filePath).toLowerCase() === '.h'
      ? fs.readFileSync(filePath, 'utf8')
      : '';
    const lang = resolveLanguageForFile(filePath, source);
    if (lang && LANG_LOADERS[lang]) found.add(lang);
  }
  return found;
}
```

### 14.7.2 Objective-C 节点提取

ObjC 的 tree-sitter grammar（`tree-sitter-objc`）将方法声明和消息发送映射为以下节点类型：

| grammar 节点类型 | 语义 | 对应图元素 |
| --- | --- | --- |
| `implementation_definition` | `@implementation ClassName` 块 | DEFINES 节点 (kind=class) |
| `method_definition` | `- (void)methodName:(Type)arg` | DEFINES 节点 (kind=method) |
| `class_method_definition` | `+ (instancetype)create` | DEFINES 节点 (kind=class_method) |
| `message_expression` | `[receiver selector:arg]` | CALLS 边 |
| `function_definition` | C 函数（`.m` 文件中混写） | DEFINES 节点 (kind=function) |

```javascript
// parser.js —— extractNodes() 对 ObjC 的处理分支
function extractObjcNodes(tree, filePath, db) {
  const defs = [];
  const calls = [];

  function walk(node) {
    switch (node.type) {
      case 'implementation_definition': {
        // 类名在第一个 named child（通常是 `type_identifier`）
        const nameNode = node.namedChildren.find(c => c.type === 'type_identifier');
        const className = nameNode?.text ?? '<anon>';
        defs.push({
          name: className,
          kind: 'class',
          file_path: filePath,
          start_line: node.startPosition.row + 1,
          end_line: node.endPosition.row + 1,
          signature: `@implementation ${className}`,
        });
        break;
      }
      case 'method_definition':
      case 'class_method_definition': {
        // ObjC 方法名由多段 selector 拼接：methodName:withParam: → "methodName:withParam:"
        const selector = buildObjcSelector(node);
        defs.push({
          name: selector,
          kind: node.type === 'class_method_definition' ? 'class_method' : 'method',
          file_path: filePath,
          start_line: node.startPosition.row + 1,
          end_line: node.endPosition.row + 1,
          signature: node.text.split('\n')[0].slice(0, 120),
        });
        break;
      }
      case 'message_expression': {
        // `[receiver selector:arg]` → 提取 selector 作为被调用目标名
        const receiver = node.namedChildren[0]?.text ?? '?';
        const sel = buildObjcSelectorFromMessage(node);
        calls.push({ caller_file: filePath, callee_name: sel, receiver_hint: receiver });
        break;
      }
    }
    for (const child of node.namedChildren) walk(child);
  }

  walk(tree.rootNode);
  return { defs, calls };
}

/** 从 method_definition 节点拼接 ObjC selector，例如 "doThing:withParam:" */
function buildObjcSelector(methodNode) {
  const parts = [];
  for (const child of methodNode.namedChildren) {
    if (child.type === 'method_selector') {
      // method_selector 的文本即 `name:` 段
      parts.push(child.text.replace(/\s+/g, ''));
    }
  }
  return parts.join('') || methodNode.namedChildren.find(c => c.type === 'identifier')?.text ?? '<anon>';
}

/** 从 message_expression 提取 selector */
function buildObjcSelectorFromMessage(msgNode) {
  const parts = [];
  for (const child of msgNode.namedChildren.slice(1)) { // 跳过 receiver
    if (child.type === 'keyword_argument') {
      const kw = child.namedChildren.find(c => c.type === 'identifier');
      if (kw) parts.push(kw.text + ':');
    }
  }
  return parts.join('') || msgNode.namedChildren[1]?.text ?? '<unknown>';
}
```

**关键设计**：

- ObjC selector 是拼接字符串，`"performTarget:action:params:"` 与 Python 中的 `performTarget:action:params:` 语义等价，作为 `name` 存入图节点。
- `message_expression` 只提取 callee name（selector），不解析 receiver 类型（需要类型系统才能静态解析）。将 `receiver_hint` 存入 edge metadata 供后处理增强。
- CTMediator 模式：`[self performTarget:@"Trade" action:@"trade_checkoutJumpToBuySellVC" params:params]` 的 selector 为 `performTarget:action:params:`，由 §14.8 适配层在后处理阶段解析字符串字面量提取 action_id。

### 14.7.3 Swift 节点提取

`tree-sitter-swift` grammar 将 Swift 源码映射为以下关键节点类型：

| grammar 节点类型 | 语义 | 对应图元素 |
| --- | --- | --- |
| `function_declaration` | `func foo()` / `func foo() async` | DEFINES 节点 (kind=function/method) |
| `class_declaration` | `class Foo: Bar` | DEFINES 节点 (kind=class) |
| `struct_declaration` | `struct Foo` | DEFINES 节点 (kind=struct) |
| `protocol_declaration` | `protocol FooProtocol` | DEFINES 节点 (kind=interface) |
| `call_expression` | `foo(args)` | CALLS 边 |
| `navigation_expression` | `receiver.method(args)` | CALLS 边（链式调用） |

```javascript
// parser.js —— extractSwiftNodes()
function extractSwiftNodes(tree, filePath) {
  const defs = [];
  const calls = [];

  function walk(node) {
    switch (node.type) {
      case 'function_declaration': {
        const nameNode = node.namedChildren.find(c => c.type === 'simple_identifier');
        const name = nameNode?.text ?? '<anon>';
        // async / throws 修饰
        const modifiers = node.namedChildren
          .filter(c => c.type === 'function_modifiers' || c.type === 'async' || c.type === 'throws')
          .map(c => c.text).join(' ');
        defs.push({
          name,
          kind: 'function',
          file_path: filePath,
          start_line: node.startPosition.row + 1,
          end_line: node.endPosition.row + 1,
          signature: `${modifiers ? modifiers + ' ' : ''}func ${name}`,
        });
        break;
      }
      case 'class_declaration':
      case 'struct_declaration':
      case 'protocol_declaration': {
        const kindMap = {
          class_declaration: 'class',
          struct_declaration: 'struct',
          protocol_declaration: 'interface',
        };
        const nameNode = node.namedChildren.find(c => c.type === 'type_identifier');
        defs.push({
          name: nameNode?.text ?? '<anon>',
          kind: kindMap[node.type],
          file_path: filePath,
          start_line: node.startPosition.row + 1,
          end_line: node.endPosition.row + 1,
          signature: node.text.split('\n')[0].slice(0, 120),
        });
        break;
      }
      case 'call_expression': {
        // call_expression: [function, argument_list]
        const callee = node.namedChildren[0];
        if (callee?.type === 'simple_identifier') {
          calls.push({ caller_file: filePath, callee_name: callee.text });
        } else if (callee?.type === 'navigation_expression') {
          // receiver.method → 取最后一段
          const parts = callee.text.split('.');
          calls.push({ caller_file: filePath, callee_name: parts[parts.length - 1] });
        }
        break;
      }
    }
    for (const child of node.namedChildren) walk(child);
  }

  walk(tree.rootNode);
  return { defs, calls };
}
```

**关键设计**：

- Swift `async/await` 函数和普通函数在图中均作为 `kind=function` 处理，`signature` 保留 `async` 修饰词供 review 层感知。
- `@objcMembers` / `@objc` 注解由适配层在后处理中标记，不影响核心解析。
- `CTMediator.sharedInstance().performTarget(...)` 通过 `call_expression` + `navigation_expression` 提取，`callee_name = performTarget`，后续由 §14.8 解析字符串参数。

### 14.7.4 Kotlin 节点提取

`tree-sitter-kotlin` grammar 关键节点类型：

| grammar 节点类型 | 语义 | 对应图元素 |
| --- | --- | --- |
| `function_declaration` | `fun foo()` / `suspend fun foo()` | DEFINES 节点 (kind=function) |
| `class_declaration` | `class Foo` / `object Foo` | DEFINES 节点 (kind=class) |
| `call_expression` | `foo(args)` | CALLS 边 |
| `navigation_expression` | `obj.method()` | CALLS 边 |

```javascript
// parser.js —— extractKotlinNodes()
function extractKotlinNodes(tree, filePath) {
  const defs = [];
  const calls = [];

  function walk(node) {
    switch (node.type) {
      case 'function_declaration': {
        const nameNode = node.namedChildren.find(c => c.type === 'simple_identifier');
        const name = nameNode?.text ?? '<anon>';
        // suspend / override / internal 修饰词
        const modifiers = [];
        for (const child of node.namedChildren) {
          if (child.type === 'modifiers') {
            for (const mod of child.namedChildren) {
              if (mod.type !== 'annotation') modifiers.push(mod.text);
            }
          }
        }
        // @HiddenFromObjC 注解 → 标记为内部跨平台接口
        const annotations = [];
        for (const child of node.namedChildren) {
          if (child.type === 'modifiers') {
            for (const mod of child.namedChildren) {
              if (mod.type === 'annotation') annotations.push(mod.text);
            }
          }
        }
        const isCrossplatform = annotations.some(a =>
          /@FlowInterop|@HiddenFromObjC|@ObjCName/.test(a)
        );
        defs.push({
          name,
          kind: 'function',
          file_path: filePath,
          start_line: node.startPosition.row + 1,
          end_line: node.endPosition.row + 1,
          signature: `${modifiers.join(' ')} fun ${name}`,
          meta: isCrossplatform ? { crossplatform: true } : {},
        });
        break;
      }
      case 'class_declaration': {
        const nameNode = node.namedChildren.find(c => c.type === 'type_identifier');
        defs.push({
          name: nameNode?.text ?? '<anon>',
          kind: 'class',
          file_path: filePath,
          start_line: node.startPosition.row + 1,
          end_line: node.endPosition.row + 1,
          signature: node.text.split('\n')[0].slice(0, 120),
        });
        break;
      }
      case 'call_expression': {
        const callee = node.namedChildren[0];
        const calleeName = callee?.type === 'navigation_expression'
          ? callee.text.split('.').pop()
          : callee?.text ?? '<unknown>';
        calls.push({ caller_file: filePath, callee_name: calleeName });
        break;
      }
    }
    for (const child of node.namedChildren) walk(child);
  }

  walk(tree.rootNode);
  return { defs, calls };
}
```

**关键设计**：

- `@HiddenFromObjC` / `@FlowInterop.Disabled` 标注的函数是 KMP 跨平台公共接口，`meta.crossplatform = true` 供 §14.8 入口检测策略使用。
- `suspend fun` 在图中等同于普通函数，`modifiers` 中含 `suspend` 关键词，不影响调用边提取。
- Kotlin object 声明（`object Foo`）的 `class_declaration` 节点中不含构造函数，图中作为 `kind=class` 处理。

## 14.8 iOS 架构适配层

> **设计原则**：核心算法（§14.1–§14.6）保持语言无关，iOS 特定逻辑通过**策略模式（Strategy Pattern）**注入。适配层不修改核心算法内部实现，只在外部提供配置对象和钩子函数。

### 14.8.1 iOS 项目自动检测

```javascript
// adapters/ios/detect.js
const fs = require('node:fs');
const path = require('node:path');

/**
 * 检测目标仓库是否为 CocoaPods 管理的 iOS 项目。
 * 判据：根目录存在 Podfile.lock，且 Pods/ 目录存在。
 *
 * @param {string} repoRoot - 目标仓库绝对路径
 * @returns {{ isIos: boolean, podlockPath: string|null }}
 */
function detectIosProject(repoRoot) {
  const podlockPath = path.join(repoRoot, 'Podfile.lock');
  const podsDir = path.join(repoRoot, 'Pods');
  const hasPodlock = fs.existsSync(podlockPath);
  const hasPodsDir = fs.existsSync(podsDir);
  return {
    isIos: hasPodlock && hasPodsDir,
    // hasPods 已移除：其值恒等于 isIos，无独立语义
    podlockPath: hasPodlock ? podlockPath : null,
  };
}

/**
 * 从 Podfile.lock 解析本地（内部业务）Pod 名称列表。
 *
 * 判据：EXTERNAL SOURCES 段中有 `:path:` 声明的 Pod 为本地 Pod。
 * 这是 CocoaPods 唯一可靠的本地 Pod 标识，不依赖命名约定（如 HS* 前缀）。
 *
 * Podfile.lock EXTERNAL SOURCES 格式：
 * ```
 * EXTERNAL SOURCES:
 *   HSTradeModule:
 *     :path: ../HSTradeModule       ← 有 :path: = 本地 Pod
 *   AFNetworking:
 *     :git: https://...             ← 无 :path: = 非本地 Pod
 * ```
 *
 * 性能：Podfile.lock 通常 < 200KB，一次性读取解析无问题。
 *
 * @param {string} podlockPath
 * @returns {string[]} 本地 Pod 名称列表，例如 ["HSTradeModule", "HSUserModule"]
 */
function parseInternalPods(podlockPath) {
  const content = fs.readFileSync(podlockPath, 'utf8');
  const localPods = new Set();

  // 定位 EXTERNAL SOURCES 段（顶级 YAML key，到下一个顶级 key 或文件末尾为止）
  const match = content.match(/^EXTERNAL SOURCES:\n([\s\S]*?)(?=^\S|\s*$)/m);
  if (!match) return [];

  let currentPod = null;
  for (const line of match[1].split('\n')) {
    // Pod 名行：2 个空格缩进，行尾 ":"
    const podNameMatch = line.match(/^  (\S+):$/);
    if (podNameMatch) { currentPod = podNameMatch[1]; continue; }

    // :path: 行：4 个空格缩进（本地路径声明）
    if (currentPod && /^    :path:/.test(line)) localPods.add(currentPod);
  }

  return [...localPods];
}

/**
 * 计算 Pods 目录的动态排除路径。
 *
 * 通过 parseInternalPods 精确区分外部三方 Pod 与内部业务 Pod，
 * 返回值注入 collectInputFiles(repoRoot, { extraExcludes, extraIncludes })，
 * 用户无需在 .spec-first-graphignore 中手动维护 Pod 命名约定白名单。
 *
 * @param {string} repoRoot
 * @param {string} podlockPath
 * @returns {{ extraExcludes: string[], extraIncludes: string[] }}
 */
function computePodExcludePaths(repoRoot, podlockPath) {
  const localPods = parseInternalPods(podlockPath);
  if (localPods.length === 0) {
    // 无本地 Pod → 排除整个 Pods/ 目录（等同于 Pods/** 规则）
    return { extraExcludes: ['Pods/**'], extraIncludes: [] };
  }

  const podsDir = path.join(repoRoot, 'Pods');
  let allPodDirs;
  try {
    allPodDirs = fs.readdirSync(podsDir, { withFileTypes: true })
      .filter(d => d.isDirectory())
      .map(d => d.name);
  } catch {
    // Pods/ 目录不存在或无读权限 → 安全降级：排除整个 Pods/
    return { extraExcludes: ['Pods/**'], extraIncludes: [] };
  }

  const localSet = new Set(localPods);
  return {
    // 外部三方 Pod：按目录精确排除（不排除本地 Pod 目录）
    extraExcludes: allPodDirs.filter(d => !localSet.has(d)).map(d => `Pods/${d}/**`),
    // 本地业务 Pod：显式标记为 include（便于日志说明）
    extraIncludes: localPods.map(p => `Pods/${p}/**`),
  };
}

module.exports = { detectIosProject, parseInternalPods, computePodExcludePaths };
```

**集成点**：`crg build` 主流程在初始化时调用 `detectIosProject(repoRoot)`，若 `isIos === true` 则加载 iOS 适配层，覆盖默认的入口检测策略和社区检测策略。

### 14.8.2 CTMediator 入口检测策略

> **背景**：UIKit 生命周期回调（`viewDidLoad`、`viewWillAppear`、`dealloc`）在业务调用图中 fan-in=0，导致入口集膨胀到 18,000 个。华盛证券项目真实业务流约 150 条，过滤依赖语义识别，不能依赖 fan-in。

```javascript
// adapters/ios/entry-points.js

/**
 * iOS 入口检测策略。
 *
 * 替换 §14.4 flows.js 中的默认 fan-in=0 判据，
 * 通过三类语义模式识别真实业务入口。
 *
 * @typedef {{ name: string, file_path: string, kind: string, id: number }} GraphNode
 * @param {GraphNode} node
 * @returns {{ isEntry: boolean, entryType: string|null, actionId: string|null }}
 */
function classifyEntryPoint(node) {
  // ── 模式 1：CTMediator Action_ 方法 ──────────────────────────────────────
  // ObjC：- (void)Action_trade_checkoutJumpToBuySellVC:(NSDictionary *)params
  // 文件：HSTarget_Trade.m, HSTarget_User.m, ...
  if (/^Action_/.test(node.name) && /HSTarget_/.test(node.file_path)) {
    const actionId = parseCtmediatorActionId(node.name);
    return { isEntry: true, entryType: 'ctmediator_action', actionId };
  }

  // ── 模式 2：Socket / 实时推送处理器 ──────────────────────────────────────
  // 文件路径含 Socket/Subscribe/Notify；方法名以 dispatch/didReceive/onMessage/handleNotify 开头
  if (
    /Socket|Subscribe|Notify/i.test(node.file_path) &&
    /^dispatch|^didReceive|^onMessage|^handleNotify/i.test(node.name)
  ) {
    return { isEntry: true, entryType: 'socket_handler', actionId: `socket/${node.name}` };
  }

  // ── 模式 3：KMP Kotlin 公共跨平台函数 ────────────────────────────────────
  // 条件：.kt 文件 + kind=function + 非私有/测试方法 + 且满足以下之一：
  //   a) meta.crossplatform=true（解析器从 @FlowInterop / @HiddenFromObjC 注解写入）
  //   b) 文件位于 hs-repos/biz-common 公共模块路径
  // 注：不加 (a)|(b) 约束会将所有 Kotlin 业务函数标为入口（过宽），破坏流分析语义。
  //     node.extra 由 detectFlows 的 allFuncs 查询提供（含 extra 列）。
  if (
    node.file_path.endsWith('.kt') &&
    node.kind === 'function' &&
    !/^_|^test/i.test(node.name)
  ) {
    const meta = node.extra
      ? (() => { try { return JSON.parse(node.extra); } catch { return {}; } })()
      : {};
    const isCrossplatform = meta.crossplatform === true;
    const isBizCommon = /[/\\](?:\.hs-repos|hs-repos)[/\\]biz-common[/\\]/.test(node.file_path);
    if (isCrossplatform || isBizCommon) {
      return { isEntry: true, entryType: 'kotlin_entry', actionId: `kotlin/${node.name}` };
    }
  }

  return { isEntry: false, entryType: null, actionId: null };
}

/**
 * 将 ObjC CTMediator Action_ 方法名转换为稳定 action_id。
 *
 * 转换规则：
 *   Action_<Module>_<action>  →  <module>/<action>
 *
 * 示例：
 *   Action_trade_checkoutJumpToBuySellVC → "trade/checkoutJumpToBuySellVC"
 *   Action_User_getUserInfo              → "user/getUserInfo"
 *
 * 大小写归一化：module 首字母小写（与 CTMediator performTarget 字符串一致）。
 *
 * @param {string} methodName - ObjC 方法 selector，例如 "Action_trade_checkoutJumpToBuySellVC:"
 * @returns {string} action_id
 */
function parseCtmediatorActionId(methodName) {
  // 去掉末尾冒号（ObjC selector）和参数类型
  const bare = methodName.replace(/:.*$/, '').replace(/^Action_/, '');
  const underscoreIdx = bare.indexOf('_');
  if (underscoreIdx === -1) return bare.toLowerCase();
  const module = bare.slice(0, underscoreIdx);
  const action = bare.slice(underscoreIdx + 1);
  return `${module.charAt(0).toLowerCase()}${module.slice(1)}/${action}`;
}

module.exports = { classifyEntryPoint, parseCtmediatorActionId };
```

**关键设计**：

- `action_id` 语义稳定：`trade/checkoutJumpToBuySellVC` 在重命名文件、迁移 Pod、重建图后保持不变（因为 CTMediator 字符串在代码中是硬编码的业务标识符）。对比 `symbol_key`（含文件路径+行号），`action_id` 不随代码结构变化而失效。
- 模式 1 依赖文件名模式 `HSTarget_*.m`，不仅依赖方法名前缀，避免将测试辅助方法中的 `Action_*` 误判为业务入口。
- 模式 3 对 Kotlin 公共函数保守：排除私有（`_` 前缀）和测试（`test` 前缀）方法，适合 KMP biz-common 层。

### 14.8.3 Pod 边界社区检测

> **背景**：3-Pass 社区检测（§14.5）在 11,400 文件的 iOS 项目上生成 ~900 个碎片社区，架构意义低。CocoaPods 的 `Pods/HS*/` 目录结构已经提供了业务方认可的模块边界，约 100 个 HS* Pod 对应约 100 个有名有姓的业务模块。

```javascript
// adapters/ios/pod-communities.js
const path = require('node:path');

/**
 * 基于 CocoaPods 目录结构提取 Pod 边界社区。
 *
 * 算法复杂度：O(N)，N 为节点数。
 * 社区数量：~本地 Pod 数量（通常 ~100），远少于 3-Pass 的 ~900。
 *
 * @param {import('better-sqlite3').Database} db
 * @param {Set<string>} localPodNames - 本地 Pod 名称集合（来自 parseInternalPods）
 * @returns {Map<string, Set<number>>}  communityName → nodeId Set
 */
function buildPodCommunities(db, localPodNames) {
  const rows = db.prepare(`
    SELECT id, file_path
    FROM nodes
    WHERE kind IN ('function', 'method', 'class', 'struct', 'interface', 'module')
  `).all();

  const groups = new Map();

  for (const { id, file_path } of rows) {
    const community = classifyToPod(file_path, localPodNames);
    if (!groups.has(community)) groups.set(community, new Set());
    groups.get(community).add(id);
  }

  return groups;
}

/**
 * 根据文件路径确定 Pod 归属。
 *
 * 优先级（高→低）：
 *   1. Pods/<localPodName>/ → 使用 Pod 名（数据驱动，不依赖命名约定）
 *   2. hs-repos/biz-common  → 归入 "KMP_BizCommon"（KMP 公共模块）
 *   3. 其他                 → 归入 "MainApp"
 *
 * @param {string} filePath       - 节点文件路径（相对于 repoRoot）
 * @param {Set<string>} localPodNames - 本地 Pod 名称集合（来自 parseInternalPods）
 * @returns {string} 社区名
 */
function classifyToPod(filePath, localPodNames) {
  // 本地 Pod：路径含 /Pods/<podName>/，且 podName 在本地 Pod 集合中
  const podMatch = filePath.match(/\/Pods\/([^/]+)\//);
  if (podMatch && localPodNames.has(podMatch[1])) return podMatch[1];

  // KMP biz-common（.hs-repos/ 或 hs-repos/ 子目录）
  if (/[/\\](?:\.hs-repos|hs-repos)[/\\]biz-common[/\\]/.test(filePath)) {
    return 'KMP_BizCommon';
  }

  // 主 App 代码
  return 'MainApp';
}

/**
 * 计算 Pod 社区健康度（O(E) 单次扫描，替代 §14.5 的 O(C×E)）。
 *
 * @param {Map<string, Set<number>>} communities - buildPodCommunities 输出
 * @param {{ source_id: number, target_id: number }[]} allEdges - 全量边列表
 * @returns {Map<string, { intra: number, inter: number, cohesion: number }>}
 */
function evaluatePodCommunityHealth(communities, allEdges) {
  // 1. 构建 nodeId → communityName 反向映射（O(N)）
  const nodeComm = new Map();
  for (const [name, nodeIds] of communities) {
    for (const id of nodeIds) nodeComm.set(id, name);
  }

  // 2. 单次边扫描（O(E)），累积 intra/inter 计数
  const counts = new Map();
  for (const name of communities.keys()) {
    counts.set(name, { intra: 0, inter: 0 });
  }

  for (const { source_id, target_id } of allEdges) {
    const sc = nodeComm.get(source_id);
    const tc = nodeComm.get(target_id);
    if (sc == null && tc == null) continue;

    if (sc === tc && sc != null) {
      counts.get(sc).intra++;
    } else {
      if (sc != null) counts.get(sc).inter++;
      if (tc != null) counts.get(tc).inter++;
    }
  }

  // 3. 计算内聚度：cohesion = intra / (intra + inter)，intra=0 时为 0
  const result = new Map();
  for (const [name, { intra, inter }] of counts) {
    const total = intra + inter;
    result.set(name, { intra, inter, cohesion: total > 0 ? intra / total : 0 });
  }
  return result;
}

/**
 * 将 Pod 社区写入 DB，接口与 §14.5 buildCommunities() 的写入段完全一致。
 *
 * 此函数是 iOS 适配层与核心主流程的适配器（adaptor）：
 * buildPodCommunities 返回 Map，但主流程（§14.8.5）通过 adapter.buildCommunities(db)
 * 调用后直接写入 DB——此函数完成 Map → DB rows 的转换，保持接口统一。
 *
 * @param {import('better-sqlite3').Database} db
 * @param {Set<string>} localPodNames - 本地 Pod 名称集合（来自 parseInternalPods）
 * @returns {{ total: number, by_status: Record<string, number> }}
 */
function buildAndWritePodCommunities(db, localPodNames) {
  const groups = buildPodCommunities(db, localPodNames);
  const allEdges = db.prepare('SELECT source_id, target_id FROM edges').all();
  const healthMap = evaluatePodCommunityHealth(groups, allEdges);

  const insertCommunity     = db.prepare(`
    INSERT INTO communities (name, file_count, health_status, health_density, health_independence)
    VALUES (?, ?, ?, ?, ?)
  `);
  const updateNodeCommunity = db.prepare('UPDATE nodes SET community_id = ? WHERE id = ?');

  const byStatus = { healthy: 0, isolated: 0, scattered: 0, fragmented: 0 };

  db.transaction(() => {
    // 外键安全顺序（与 §14.5 buildCommunities 一致）：
    // 先 NULL nodes.community_id，再删旧社区，再插入新社区并回填
    db.prepare('UPDATE nodes SET community_id = NULL').run();
    db.prepare('DELETE FROM communities').run();
    for (const [name, nodeIds] of groups) {
      const h = healthMap.get(name) ?? { cohesion: 0 };
      // Pod 社区使用 independence = cohesion（相同语义，字段名沿用通用表）
      const status = h.cohesion >= 0.5 ? 'healthy' : h.cohesion >= 0.2 ? 'isolated' : 'fragmented';
      byStatus[status]++;
      const { lastInsertRowid: cid } = insertCommunity.run(
        name, nodeIds.size, status,
        +(h.cohesion.toFixed(3)), +(h.cohesion.toFixed(3))
      );
      for (const nid of nodeIds) updateNodeCommunity.run(cid, nid);
    }
    // Pass 4：community_id 传播到同文件的非 module 节点（与通用算法保持一致）
    db.prepare(`
      UPDATE nodes
      SET community_id = (
        SELECT m.community_id FROM nodes m
        WHERE m.kind = 'module' AND m.file_path = nodes.file_path
        LIMIT 1
      )
      WHERE kind != 'module' AND community_id IS NULL
    `).run();
  })();

  return { total: groups.size, by_status: byStatus };
}

module.exports = { buildPodCommunities, classifyToPod, evaluatePodCommunityHealth, buildAndWritePodCommunities };
```

**关键设计**：

- `buildPodCommunities` 直接从文件路径提取，不需要图遍历，O(N) 时间复杂度。
- `evaluatePodCommunityHealth` 复用 §14.5 中证明过的 O(E) 单次扫描算法，结果与逐社区扫描数学等价（参见 §14.5 中的四种 case 分析）。
- `buildAndWritePodCommunities` 是 Map → DB 的适配函数，与 §14.5 `buildCommunities()` 的写入接口保持一致，主流程可统一调用 `adapter.buildCommunities(db)`（见 §14.8.5）。
- 社区数量从 ~900 降至 ~100，使 `get_architecture_overview` 和 `list_communities` 等工具的输出实际可读。

### 14.8.4 action_id 稳定寻址与 3-Tier 符号解析

> **背景**：iOS CTMediator 项目中，同名 ObjC 方法（如 `init`、`setup`、`dealloc`）可能出现在数百个类中，使用 bare name 寻址会返回大量候选。`symbol_key`（含行号）跨 rerun 失效。`action_id` 提供语义稳定的第三条路。

```javascript
// adapters/ios/symbol-resolver.js

/**
 * 3-Tier 符号解析器。
 *
 * 层级（优先级高→低）：
 *   L1：action_id      - "trade/checkoutJumpToBuySellVC"（CTMediator 专属，语义稳定）
 *   L2：file#name      - "HSTarget_Trade.m#Action_trade_checkoutJumpToBuySellVC"
 *   L3：bare name      - "checkoutJumpToBuySellVC"（可能返回候选列表）
 *
 * 原则：永不静默选第一个。遇到歧义必须返回候选列表，让调用方或用户决策。
 *
 * @param {import('better-sqlite3').Database} db
 * @param {string} input - 用户输入的符号标识符
 * @returns {{ tier: 'L1'|'L2'|'L3', match: 'exact'|'qualified'|'name'|'ambiguous'|'not_found', results: object[] }}
 */
function resolveSymbol(db, input) {
  // ── L1：action_id 格式 "module/action" ────────────────────────────────────
  if (/^[a-z][a-zA-Z0-9]*\/[a-zA-Z0-9_]+$/.test(input)) {
    const rows = db.prepare(`
      SELECT n.*, f.action_id
      FROM nodes n
      JOIN flows f ON f.entry_point_id = n.id
      WHERE f.action_id = ?
    `).all(input);
    if (rows.length === 1) return { tier: 'L1', match: 'exact', results: rows };
    if (rows.length > 1) return { tier: 'L1', match: 'ambiguous', results: rows };
    // action_id 未命中则降至 L3（可能是 bare name 包含斜杠的边缘情况）
  }

  // ── L2：file#name 格式（文件名片段 + # + 符号名）──────────────────────────
  if (input.includes('#')) {
    const [filePart, namePart] = input.split('#', 2);
    const rows = db.prepare(`
      SELECT * FROM nodes
      WHERE name = ? AND file_path LIKE ?
      ORDER BY start_line ASC
    `).all(namePart, `%${filePart}%`);
    if (rows.length === 1) return { tier: 'L2', match: 'qualified', results: rows };
    if (rows.length > 1) return { tier: 'L2', match: 'ambiguous', results: rows };
    if (rows.length === 0) return { tier: 'L2', match: 'not_found', results: [] };
  }

  // ── L3：bare name ─────────────────────────────────────────────────────────
  const rows = db.prepare(`
    SELECT * FROM nodes
    WHERE name = ?
    ORDER BY file_path ASC, start_line ASC
  `).all(input);

  if (rows.length === 0) return { tier: 'L3', match: 'not_found', results: [] };
  if (rows.length === 1) return { tier: 'L3', match: 'name', results: rows };

  // ── 歧义：返回候选列表（永不静默选第一个）────────────────────────────────
  return {
    tier: 'L3',
    match: 'ambiguous',
    results: rows,
    hint: `发现 ${rows.length} 个同名符号，请用 file#name 或 action_id 精确定位：\n` +
      rows.slice(0, 5).map(r => `  ${path.basename(r.file_path)}#${r.name}  (${r.file_path}:${r.start_line})`).join('\n'),
  };
}

module.exports = { resolveSymbol };
```

**关键设计**：

- L1 `action_id` 基于 CTMediator 字符串字面量，不含路径和行号，跨 rerun 天然稳定。
- L2 `file#name` 模糊匹配文件名片段（`LIKE %HSTarget_Trade.m%`），容忍路径变化，精确到文件内唯一方法。
- L3 bare name 歧义时返回前 5 个候选并给出 `hint`，用户或 LLM 调用方可据此升级到 L2/L1。
- **绝不静默返回第一个**：避免 LLM 在歧义情况下基于错误上下文生成幻觉分析，这是与朴素 `indexOf` 实现的关键区别。

### 14.8.5 适配层集成方式

适配层通过**配置对象注入**接入核心算法，不修改 §14.1–§14.6 的内部实现。

```javascript
// adapters/ios/index.js —— iOS 适配层统一出口
//
// 职责：提供 iOS 项目专用的"策略"（入口检测、社区划分、符号解析）。
// 不负责语言检测——语言由 §14.7.1 detectPresentLanguages(finalInputs) 从输入收敛结果推导，两件事独立。

const { detectIosProject, parseInternalPods, computePodExcludePaths } = require('./detect');
const { classifyEntryPoint } = require('./entry-points');
const { buildAndWritePodCommunities } = require('./pod-communities');
const { resolveSymbol } = require('./symbol-resolver');

/**
 * 构建 iOS 适配器配置对象。
 * 注入到 crg build 主流程的 AdapterConfig 槽位。
 *
 * @param {string} repoRoot
 * @returns {AdapterConfig|null} 若非 iOS 项目返回 null（使用默认策略）
 */
function buildIosAdapter(repoRoot) {
  const { isIos, podlockPath } = detectIosProject(repoRoot);
  if (!isIos) return null;

  // 解析本地 Pod 列表（数据驱动，不依赖命名约定）
  const localPods = parseInternalPods(podlockPath);
  const localPodNames = new Set(localPods);

  // ── 输入模式自动升级 ───────────────────────────────────────────────────────
  // 本地 Pod 通常被 .gitignore 忽略，tracked-only 模式下 Pods/ 不在 git ls-files 候选集。
  // extraIncludes 是过滤器（从候选集中保留），无法将未跟踪文件注入候选集。
  // 因此：若存在本地 Pod，自动升级为 tracked+untracked，并输出 warning。
  //
  // 如不希望自动升级，可在 CLI 层传入 --mode=tracked-only 覆盖（仅影响本地 Pod 入图，
  // 不影响三方 Pod 的排除逻辑）。
  const requiresUntrackedMode = localPods.length > 0;
  if (requiresUntrackedMode) {
    process.stderr.write(
      `[crg] WARNING: iOS project with ${localPods.length} local Pod(s) detected. ` +
      `Upgrading input mode to "tracked+untracked" so local Pod sources can be indexed. ` +
      `Pass --mode=tracked-only to disable.\n`
    );
  }

  return {
    // ── 输入收敛：动态排除外部三方 Pod，精确保留内部业务 Pod ─────────────
    // 注入 collectInputFiles(repoRoot, { extraExcludes, extraIncludes, mode })。
    // 用户无需在 .spec-first-graphignore 维护 Pod 命名约定白名单。
    podExcludes: computePodExcludePaths(repoRoot, podlockPath),
    // mode 升级：本地 Pod 存在时切换为 tracked+untracked，使 Pods/<localPod>/ 文件进入候选集
    inputMode: requiresUntrackedMode ? 'tracked+untracked' : undefined,  // undefined = 使用调用方默认值

    // ── 入口检测策略：替换 flows.js 的 fan-in=0 判据 ─────────────────────
    isEntryPoint: classifyEntryPoint,

    // ── 社区检测策略：替换 3-Pass communities.js ──────────────────────────
    // localPodNames 从 Podfile.lock 推导，classifyToPod 不再硬编码 HS 前缀
    buildCommunities: (db) => buildAndWritePodCommunities(db, localPodNames),

    // ── 符号解析策略：为 `crg flow <name>` 提供 L1/L2/L3 解析 ────────────
    resolveSymbol: (db, input) => resolveSymbol(db, input),

    // 注意：不包含 langExtensions。
    // 语言由 detectPresentLanguages(finalInputs) 从输入收敛结果推导，与项目类型无关。
  };
}

module.exports = { buildIosAdapter };
```

**主流程集成**（`crg build` / `crg update` 入口）：

```javascript
// cli.js / build.js —— 主流程伪代码
const { detectPresentLanguages } = require('./parser');
const { collectInputFiles }      = require('./input-selection');
const { buildIosAdapter }        = require('./adapters/ios');

async function runBuild(repoRoot, options) {
  // 1. 项目类型检测（必须在输入收敛前：adapter.podExcludes 影响输入集合）
  const adapter = buildIosAdapter(repoRoot) ?? DEFAULT_ADAPTER;

  // 2. 输入收敛：默认 tracked-only；iOS 项目有本地 Pod 时自动升级为 tracked+untracked
  //    adapter.podExcludes 由 Podfile.lock EXTERNAL SOURCES 精确计算
  //    adapter.inputMode 由 buildIosAdapter 在本地 Pod 存在时注入（见 §14.8.5）
  //    注：extraIncludes 是过滤器（从候选集中保留），无法注入未跟踪文件；
  //        必须先通过 inputMode 升级候选集，extraIncludes 才能生效
  const finalInputs = collectInputFiles(repoRoot, {
    ...options,
    mode: adapter.inputMode ?? options.mode ?? 'tracked-only',
    extraExcludes: adapter.podExcludes?.extraExcludes ?? [],
    extraIncludes: adapter.podExcludes?.extraIncludes ?? [],
  });

  // 3. 语言检测：从最终输入集合推导需要加载的 grammar（不再全盘扫描）
  const langs = detectPresentLanguages(finalInputs);
  // 例：前后端 monorepo → Set { 'ts', 'tsx', 'go' }
  // 例：iOS 项目       → Set { 'swift', 'm', 'kt' }
  // 例：RN 项目        → Set { 'ts', 'tsx', 'swift', 'm', 'kt' }

  // 4. 解析阶段：对 finalInputs 进行解析
  await parseFiles(finalInputs, langs, db);

  // 5. 后处理阶段：使用适配器的社区和入口策略
  await adapter.buildCommunities(db);   // 已含 DB 写入，返回 { total, by_status }
  const flows = detectFlows(db, { isEntryPoint: adapter.isEntryPoint });
  await writeFlows(db, flows);
}
