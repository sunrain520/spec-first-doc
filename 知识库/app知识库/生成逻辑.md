## 1. 基础准备

```bash
cd /Users/kuang/xiaobu/hszq-app

source ~/.venvs/repowiki/bin/activate

repowiki config set language zh
repowiki config set max_files 1000
repowiki config set max_file_size 204800
```

## 3. 定义扫描函数

这样每个模块都会输出到独立目录，避免互相覆盖。

```bash
scan_module() {
  MODULE=$1
  OUT_DIR="./wiki/$MODULE"

  echo "========================================"
  echo "开始分析模块: $MODULE"
  echo "输出目录: $OUT_DIR"
  echo "========================================"

  repowiki config set output_dir "$OUT_DIR"
  repowiki scan "./$MODULE" -l zh -f html --open
}
```

你也可以直接保存成脚本：

```bash
cat > run_repowiki_hszq_app.sh <<'EOF'

set -e

cd /Users/kuang/xiaobu/hszq-app
source ~/.venvs/repowiki/bin/activate

repowiki config set language zh
repowiki config set max_files 1000
repowiki config set max_file_size 204800

cat > .repowikiignore <<'IGNORE'
.git/
.idea/
.vscode/
.DS_Store
.gradle/
build/
*/build/
out/
outputs/
captures/
*.apk
*.aab
*.aar
*.dex
*.class
*.so
jacocoreport/

repoproject.json

*.gz

.env.*

scan_module() {
  MODULE=$1
  OUT_DIR="./wiki/$MODULE"

  if [ ! -d "$MODULE" ]; then
    echo "跳过不存在的模块: $MODULE"
    return
  fi

  echo "========================================"
  echo "开始分析模块: $MODULE"
  echo "输出目录: $OUT_DIR"
  echo "========================================"

  repowiki config set output_dir "$OUT_DIR"
  repowiki scan "./$MODULE" -l zh -f html --open
}

# App 基座
scan_module app-core
scan_module core
scan_module common
scan_module platformcomm
scan_module platformview
scan_module uikit

scan_module hsconfig

# 交易

# 行情
scan_module quotes

scan_module quotes-detail
scan_module quotes-rankings
scan_module quotes-watchlist

# 资讯
scan_module news

scan_module news-watchlists
scan_module headlines
scan_module expressnews

# 用户 / 设置 / 消息
scan_module me
scan_module settings
scan_module message-center

# 财富 / IPO / ETF / 权证
scan_module wealth
scan_module ipos
scan_module hs-etf
scan_module warrant

# 搜索 / 自选 / 社区
scan_module search
scan_module searchbase
scan_module watchlist-core
scan_module community
scan_module topic
scan_module post
```

```bash

```
